微信小程序获取数据、修改数据、添加数据、删除数据的处理方式
<escape><!-- more --></escape>
### 1 获取数据
```
Page({
  data: {
    inputText: 'hi',
  },
  //页面载入时打印inputText的值
  onLoad: function (options) {
    console.log(this.data.inputText)
  }
})
```

### 2 修改数据
```
Page({
  ...
  //页面载入时修改inputText的值为hello
  onLoad: function (options) {
    this.setData({
      inputText: 'hello'
    })
  }
})
```

### 3 增加数据
```
Page({
  ...
  //页面载入时修改inputText的值为hello
  onLoad: function (options) {
    this.data.inputText= this.data.list.concat(newarray);
    this.setData({
      inputText: this.data.inputText
    })
  }
})
```
